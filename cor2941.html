using namespace std;

int main(void) {
    string Word;
    int i;
    cin >> Word;
    int alpha = (int)Word.length();
    
    for(i=0;i<(int)Word.length();i++){
        if(Word[i]=='='){
            if(Word[i-2]=='d'&&Word[i-1]=='z'){
                alpha-=2;
            }else alpha-=1;
        }
        else if(Word[i]=='-'){
            alpha-=1;
        }
        else if(Word[i]=='l'||Word[i]=='n'){
            if(Word[i+1]=='j'){
                alpha-=1;
            }
        }
    }
    
    cout << alpha;
    
    
    return 0;
}
